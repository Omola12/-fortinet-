    .reply { border-left: 3px solid #4a6; padding-left: 10px; margin: 10px 0; background: #fff; }

<body>
  <header>
    <h1>Fortinet — Admin Panel</h1>
    <p>View complaints, reply, and update status.</p>
  </header>

  <main class="row">
    <section style="flex: 1;">
      <h2>Complaints</h2>
      <div id="list" class="list"></div>
    </section>

    <section style="flex: 1.5;">
      <h2>Details</h2>
      <div id="details" class="thread">Select a complaint to view details.</div>

      <h3>Send Reply</h3>
      <form id="replyForm">
        <div class="field">
          <label>Admin Name</label>
          <input type="text" name="adminName" required/>
        </div>
        <div class="field">
          <label>Message</label>
          <textarea name="message" rows="4" required></textarea>
        </div>
        <div class="field">
          <label>Status</label>
          <select name="status">
            <option value="">(no change)</option>
            <option>open</option>
            <option>pending</option>
            <option>resolved</option>
            <option>closed</option>
          </select>
        </div>
        <button type="submit" disabled>Send Reply</button>
        <p id="replyStatus" class="status"></p>
      </form>
    </section>
  </main>

  <script>
    const listEl = document.getElementById('list');
    const detailsEl = document.getElementById('details');
    const replyForm = document.getElementById('replyForm');
    const replyStatus = document.getElementById('replyStatus');

    let complaints = [];
    let selectedId = null;

    async function fetchComplaints() {
      const res = await fetch('/api/complaints');
      complaints = await res.json();
      renderList();
    }

    function renderList() {
      listEl.innerHTML = '';
      complaints.forEach(c => {
        const card = document.createElement('div');
        card.className = 'card' + (c.id === selectedId ? ' active' : '');
        card.innerHTML = `
          <div><strong>${escapeHTML(c.subject)}</strong></div>
          <div class="meta">${escapeHTML(c.name)} • ${escapeHTML(c.email)} • ${new Date(c.createdAt).toLocaleString()}</div>
          <div class="meta">Status: ${escapeHTML(c.status)}</div>
          <div>${escapeHTML(c.message).slice(0, 140)}${c.message.length > 140 ? '…' : ''}</div>
        `;
        card.addEventListener('click', () => selectComplaint(c.id));
        listEl.appendChild(card);
      });
    }

    async function selectComplaint(id) {
      selectedId = id;
      replyForm.querySelector('button[type="submit"]').disabled = false;
      renderList();
      const res = await fetch('/api/complaints/' + id);
      const c = await res.json();
      detailsEl.innerHTML = `
        <div><strong>${escapeHTML(c.subject)}</strong></div>
        <div class="meta">From: ${escapeHTML(c.name)} &lt;${escapeHTML(c.email)}&gt; • ${new Date(c.createdAt).toLocaleString()}</div>
        <div style="margin-top:8px;">${nl2br(escapeHTML(c.message))}</div>
        <div class="meta" style="margin-top:8px;">Status: ${escapeHTML(c.status)}</div>
        <h4>Replies</h4>
        ${c.replies.length ? c.replies.map(r => `
          <div class="reply">
            <div class="meta">${escapeHTML(r.adminName)} • ${new Date(r.createdAt).toLocaleString()}</div>
            <div>${nl2br(escapeHTML(r.message))}</div>
          </div>
        `).join('') : '<div class="meta">No replies yet.</div>'}
      `;
    }

    replyForm.addEventListener('submit', async (e) => {
      e.preventDefault();
      if (!selectedId) return;
      replyStatus.textContent = 'Sending...';
      const fd = new FormData(replyForm);
      const payload = {
        adminName: fd.get('adminName').trim(),
        message: fd.get('message').trim(),
        status: fd.get('status') || undefined
      };
      try {
        const res = await fetch('/api/complaints/' + selectedId + '/replies', {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify(payload)
        });
        if (!res.ok) {
          const err = await res.json().catch(() => ({}));
          throw new Error(err.error || 'Failed to send reply');
        }
        replyForm.reset();
        replyForm.querySelector('button[type="submit"]').disabled = true;
        replyStatus.textContent = 'Reply sent.';
        await fetchComplaints();
        await selectComplaint(selectedId); // refresh thread
      } catch (err) {
        replyStatus.textContent = 'Error: ' + err.message;
      } finally {
        setTimeout(() => replyStatus.textContent = '', 2000);
      }
    });

    function escapeHTML(str) {
      return String(str)
        .replaceAll('&', '&amp;')
        .replaceAll('<', '&lt;')
        .replaceAll('>', '&gt;')
        .replaceAll('"', '&quot;')
        .replaceAll("'", '&#039;');
    }

    function nl2br(str) {
      return str.replaceAll('\n', '<br>');
    }

    fetchComplaints();
    // Optional: refresh list periodically
    setInterval(fetchComplaints, 10000);
  </script>
